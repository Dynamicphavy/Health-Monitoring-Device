// Read Body Temperature from MAX30205
float readBodyTemperature() {
  Wire.beginTransmission(MAX30205_ADDRESS);
  Wire.write(0x00); // Temperature register
  Wire.endTransmission();
  
  Wire.requestFrom(MAX30205_ADDRESS, 2);
  if (Wire.available() == 2) {
    uint8_t msb = Wire.read();
    uint8_t lsb = Wire.read();
    int16_t rawTemp = ((msb << 8) | lsb) >> 3;
    return rawTemp * 0.03125;
  }
  return 0.0;
}

// Step Counting Logic
void checkSteps() {
  int x = analogRead(xPin);
  int y = analogRead(yPin);
  int z = analogRead(zPin);
  long mag = sqrt(sq(x - 2000) + sq(y - 2000) + sq(z - 2000));

  if (mag > threshold && !stepFlag) {
    steps++;
    stepFlag = true;
    Serial.print("Step detected! Total: ");
    Serial.println(steps);
  } else if (mag < threshold - 500) {
    stepFlag = false;
  }
}

// Upload Daily Summary to Firebase
void uploadDailyData() {
  if (!firebaseReady) return;

  float avgTemp = 0;
  int avgHR = 0;

  if (dailyReadingsCount > 0) {
    avgTemp = dailySumTemp / dailyReadingsCount;
    avgHR = dailySumHR / dailyReadingsCount;
  }

  Serial.println("--- Uploading Daily Summary to Firebase ---");
  
  // Create a unique path based on time (or just incrementing ID)
  // Since we don't have NTP time here, we'll use a simple "Day_X" or append
  // Ideally, use NTP to get a date string like "2023-10-27"
  String path = "/daily_summary/" + String(millis()); 

  FirebaseJson json;
  json.set("avg_temp", avgTemp);
  json.set("avg_hr", avgHR);
  json.set("total_steps", steps);
  json.set("uptime_ms", millis());

  if (Firebase.RTDB.setJSON(&fbdo, path.c_str(), &json)) {
    Serial.println("Firebase Upload Success");
  } else {
    Serial.print("Firebase Upload Failed: ");
    Serial.println(fbdo.errorReason());
  }

  // Reset Daily Counters
  dailySumTemp = 0;
  dailySumHR = 0;
  dailyReadingsCount = 0;
  steps = 0; // Reset steps for the new day
}

// Send Data to Blynk and Update Display
void sendSensorData() {
  float tempC = readBodyTemperature();
  Blynk.virtualWrite(V2, tempC);
  Blynk.virtualWrite(V3, steps);
  
  if (beatAvg > 0) {
    Blynk.virtualWrite(V0, beatAvg);
    Blynk.virtualWrite(V1, spO2);
  }
  
  updateDisplay(tempC, beatAvg, spO2, steps);

  // --- Abnormal Vitals Alarm ---
  if (millis() - lastAlertTime > alertInterval) {
    bool alertTriggered = false;
    if (tempC > 0 && (tempC < 36.5 || tempC > 37.3)) {
      Blynk.logEvent("temp_alert", String("Abnormal Temp: ") + tempC + " C");
      alertTriggered = true;
    }
    if (beatAvg > 0 && (beatAvg < 60 || beatAvg > 100)) {
      Blynk.logEvent("hr_alert", String("Abnormal HR: ") + beatAvg + " BPM");
      alertTriggered = true;
    }
    if (alertTriggered) lastAlertTime = millis();
  }

  // --- Daily Aggregation ---
  if (tempC > 0 && beatAvg > 0) {
    dailySumTemp += tempC;
    dailySumHR += beatAvg;
    dailyReadingsCount++;
  }

  // Check if 24 hours have passed
  if (millis() - lastDailyUpload > dailyInterval) {
    uploadDailyData();
    lastDailyUpload = millis();
  }
}

void loop() {

    // ------------- TEMPERATURE READINGS -------------
    if (millis() - lastTempRead > 1000) {
        lastTempRead = millis();
        temp = mlx.readObjectTempC();
    }

    // ------------- MAX30102 READINGS ------------
    mex.check();
    uint32_t irValue = mex.getIR();
    uint32_t redValue = mex.getRed();

    if (irValue > 10000) {
        // Heart Rate
        if (checkForBeat(irValue)) {
            long now = millis()
            long delta = now - lastBeat;
            lastBeat = now;
            
            bpm = 60 / (delta / 1000.0);

            if (bpm < 255 && bpm > 20) {
                rates[rateSpot++] = (byte)bpm;
                rateSpot %= RATE_SIZE;

                beatAvg = 0;
                for (byte x = 0; x < RATE_SIZE; x++)
                    beatAvg += rates[x];
                beatAvg /= RATE_SIZE;
                heartRate = beatAvg;
            }
        }

        // SpO2

        // DC Components
        avered = avered * frate + (double)redValue * (1.0 - frate);
        aveir = aveir * frate + (double)irValue * (1.0 - frate);

        // AC Components
        sumredrms += (redValue - avered) * (redValue - avered);
        sumirrms += (irValue - aveir) * (irValue - aveir);

        if (i % Num == 0) {
            double R = (sqrt(sumredrms) / avered) / (sqrt(sumirrms) / aveir);
            int calculatedSpo2 = (int)(-45.060 * R * R + 30.354 * R + 94.845);
            ESpO2 = FSpO2 * ESpO2 + (1.0 - FSpO2) * calculatedSpo2;

            if (ESpO2 <= 100 && ESpO2 >= 80)
                spo2 = (int)ESpO2;

            sumredrms = 0;
            sumirrms = 0;
            i = 0;
        }
    } else {
        
        spo2 = 0;
    }




    // ------------- STEP COUNTING -------------
    float accelX = myIMU.readFloatAccelX();
    float accelY = myIMU.readFloatAccelY();
    float accelZ = myIMU.readFloatAccelZ();

    float accelMagnitude = sqrt(accelX * accelX + accelY * accelY + accelZ * accelZ);
    float accelWithoutGravity = abs(accelMagnitude - 1.0);

    if (accelWithoutGravity > 0.3 &&
        (millis() - lastStepTime) > MIN_STEP_INTERVAL) {
            steps++;
            lastStepTime = millis();
        }

    // ------------- OLED UPDATE -------------
    if(millis() - lastOLEDUpdate > 1000) {
        lastOLEDUpdate = millis();
        healthVitals(heartRate, spo2, temp, steps); // Displayed on OLED
    }

    delay(5); // Stability delay
